        GxB_JIT_NONE = -1
            JIT disabled at compile time.  No JIT avaiable at all.

        GxB_JIT_OFF = 0
            do not use the JIT at all: free all JIT kernels if loaded, and
            dlclose all of them (call GB_jitifyer_table_free)

        GxB_JIT_PAUSE = 1
            do not run JIT kernels but keep any loaded JIT kernels loaded

        GxB_JIT_RUN = 2
            able to run JIT kernels if already loaded; may not load or compile

        GxB_JIT_LOAD = 3
            able to load and run JIT kernels; may not compile

        GxB_JIT_ON = 4
            full JIT: able to compile, load, and run
